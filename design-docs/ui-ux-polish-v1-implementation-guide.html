Key Architecture Decisions

Floating Inspector
Absolute positioned card inside a relative wrapper around the canvas. Appears only when an agent is selected. Clean, non-intrusive, scalable.
Speed Implementation
Simple multiplier inside gameLoop: for (let i=0; i<simSpeed; i++) update(); then one draw. Fast to implement, good enough for years.
Tabs
Four tabs using show/hide divs (simplest, most reliable). Single activeTab variable + renderSidebar() function.
Time Series
Three capped arrays (trendPop, trendFood, trendAvgHunger). Sample every 30 ticks. Max 400 points. Ring-buffer style (shift when full).

function updateInspector() {
  const el = document.getElementById('inspector');
  const content = document.getElementById('inspector-content');
  if (!el || !content) return;

  if (!selectedAgent) {
    el.classList.add('hidden');
    return;
  }
  el.classList.remove('hidden');

PHASE 4 Live Metrics Computation
Add computeMetrics() and call it periodically from update() (every 20 ticks is plenty).
function computeMetrics() {
  if (agents.length === 0) {
    currentMetrics = { alive: 0, deaths: starvationDeaths, totalFood: 0, avgHunger: 0, minHunger: 0, maxHunger: 0, avgFood: 0 };
    return;
  }
  let minH = Infinity, maxH = -Infinity, sumH = 0, sumFood = 0, totalFood = 0;
  for (let a of agents) {
    minH = Math.min(minH, a.hunger);
    maxH = Math.max(maxH, a.hunger);
    sumH += a.hunger;
    sumFood += a.food;
  }
  for (let f of food) totalFood += f.amount;

  currentMetrics = {
    alive: agents.length,
    deaths: starvationDeaths,
    totalFood: Math.round(totalFood * 10) / 10,
    avgHunger: sumH / agents.length,
    minHunger: minH,
    maxHunger: maxH,
    avgFood: sumFood / agents.length
  };
}
Call computeMetrics() inside update() when tick % 20 === 0.

function drawTrendChart(canvasEl, data, color, label) {
  if (!canvasEl || data.length < 2) return;
  const ctx = canvasEl.getContext('2d');
  ctx.clearRect(0, 0, canvasEl.width, canvasEl.height);
  const w = canvasEl.width, h = canvasEl.height;
  const min = Math.min(...data), max = Math.max(...data);
  ctx.strokeStyle = color; ctx.lineWidth = 1.75;
  ctx.beginPath();
  for (let i = 0; i < data.length; i++) {
    const x = (i / (data.length - 1)) * w;
    const y = h - ((data[i] - min) / (max - min || 1)) * (h - 4);
    i === 0 ? ctx.moveTo(x, y) : ctx.lineTo(x, y);
  }
  ctx.stroke();
}
Call the three charts whenever the Trends tab is visible (or on every draw when Trends is active).
